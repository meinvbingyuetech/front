        'test' : function (){

            console.log(data); //输出aaa
        }
    }

     var tt = function () {
        this.aa();
    }

    tt.prototype.aa = function () {
        console.info('aa')
    }
    tt.prototype.bb = function () {
        console.info('bb')
    }

    var obj = new tt(); // 输出aa
    obj.aa(); // 输出aa
    obj.bb(); // 输出bb
